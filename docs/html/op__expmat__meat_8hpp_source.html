50 inline
51 bool
52 op_expmat::apply_direct(Mat<typename T1::elem_type>& out, const Base<typename T1::elem_type, T1>& expr)
53  {
54  arma_extra_debug_sigprint();
55 
56  typedef typename T1::elem_type eT;
57  typedef typename T1::pod_type T;
58 
59  if(is_op_diagmat<T1>::value)
60  {
61  out = expr.get_ref(); // force the evaluation of diagmat()
62 
63  arma_debug_check( (out.is_square() == false), "expmat(): given matrix must be square sized" );
64 
65  const uword N = (std::min)(out.n_rows, out.n_cols);
66 
67  for(uword i=0; i<N; ++i) { out.at(i,i) = std::exp( out.at(i,i) ); }
68 
69  return true;
70  }
71 
72  Mat<eT> A = expr.get_ref();
73 
74  arma_debug_check( (A.is_square() == false), "expmat(): given matrix must be square sized" );
75 
76  if(A.is_diagmat())
77  {
78  arma_extra_debug_print("op_expmat: detected diagonal matrix");
79 
80  const uword N = (std::min)(A.n_rows, A.n_cols);
81 
82  out.zeros(N,N);
83 
84  for(uword i=0; i<N; ++i) { out.at(i,i) = std::exp( A.at(i,i) ); }
85 
86  return true;
87  }
88 
89  const bool try_sympd = arma_config::optimise_sympd && sympd_helper::guess_sympd(A);
90 
91  if(try_sympd)
92  {
93  arma_extra_debug_print("op_expmat: attempting sympd optimisation");
94 
95  // if matrix A is sympd, all its eigenvalues are positive
96 
97  Col< T> eigval;
98  Mat<eT> eigvec;
99 
100  const bool eig_status = eig_sym_helper(eigval, eigvec, A, 'd', "expmat()");
101 
102  if(eig_status)
103  {
104  eigval = exp(eigval);
105 
106  out = eigvec * diagmat(eigval) * eigvec.t();
107 
108  return true;
109  }
110 
111  arma_extra_debug_print("op_expmat: sympd optimisation failed");
112 
113  // fallthrough if eigen decomposition failed
114  }
115 
116  const T norm_val = arma::norm(A, "inf");
117 
118  const double log2_val = (norm_val > T(0)) ? double(eop_aux::log2(norm_val)) : double(0);
119 
120  int exponent = int(0); std::frexp(log2_val, &exponent);
121 
122  const uword s = uword( (std::max)(int(0), exponent + int(1)) );
123 
124  A /= eT(eop_aux::pow(double(2), double(s)));
125 
126  T c = T(0.5);
127 
128  Mat<eT> E(A.n_rows, A.n_rows, fill::eye); E += c * A;
129  Mat<eT> D(A.n_rows, A.n_rows, fill::eye); D -= c * A;
130 
131  Mat<eT> X = A;
132 
133  bool positive = true;
134 
135  const uword N = 6;
136 
137  for(uword i = 2; i <= N; ++i)
138  {
139  c = c * T(N - i + 1) / T(i * (2*N - i + 1));
140 
141  X = A * X;
142 
143  E += c * X;
144 
145  if(positive) { D += c * X; } else { D -= c * X; }
146 
147  positive = (positive) ? false : true;
148  }
149 
150  if( (D.is_finite() == false) || (E.is_finite() == false) ) { return false; }
151 
152  const bool status = solve(out, D, E, solve_opts::no_approx);
153 
154  if(status == false) { return false; }
155 
156  for(uword i=0; i < s; ++i) { out = out * out; }
157 
158  return true;
159  }

234
