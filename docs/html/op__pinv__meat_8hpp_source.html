87 template<typename T1>
88 inline
89 bool
90 op_pinv::apply_direct(Mat<typename T1::elem_type>& out, const Base<typename T1::elem_type,T1>& expr, typename T1::pod_type tol, const uword method_id)
91  {
92  arma_extra_debug_sigprint();
93 
94  typedef typename T1::elem_type eT;
95  typedef typename T1::pod_type T;
96 
97  arma_debug_check((tol < T(0)), "pinv(): tolerance must be >= 0");
98 
99  // method_id = 0 -> default setting
100  // method_id = 1 -> use standard algorithm
101  // method_id = 2 -> use divide and conquer algorithm
102 
103  Mat<eT> A(expr.get_ref());
104 
105  if(A.is_empty()) { out.set_size(A.n_cols,A.n_rows); return true; }
106 
107  if(is_op_diagmat<T1>::value || A.is_diagmat())
108  {
109  arma_extra_debug_print("op_pinv: detected diagonal matrix");
110 
111  return op_pinv::apply_diag(out, A, tol);
112  }
113 
114  bool do_sym = false;
115  bool do_sympd = false;
116 
117  const bool is_sym_size_ok = (A.n_rows > (is_cx<eT>::yes ? uword(20) : uword(40)));
118  const bool is_arg_default = ((tol == T(0)) && (method_id == uword(0)));
119 
120  if( (arma_config::optimise_sympd) && (auxlib::crippled_lapack(A) == false) && (is_arg_default || is_sym_size_ok) )
121  {
122  bool is_approx_sym = false;
123  bool is_approx_sympd = false;
124 
125  sympd_helper::analyse_matrix(is_approx_sym, is_approx_sympd, A);
126 
127  do_sym = is_sym_size_ok && ((is_cx<eT>::no) ? (is_approx_sym) : (is_approx_sym && is_approx_sympd));
128  do_sympd = is_arg_default && is_approx_sympd;
129  }
130 
131  if(do_sympd)
132  {
133  arma_extra_debug_print("op_pinv: attempting sympd optimisation");
134 
135  out = A;
136 
137  bool is_sympd_junk = false;
138  T rcond_calc = T(0);
139  const T rcond_threshold = T((std::max)(uword(100), uword(A.n_rows))) * std::numeric_limits<T>::epsilon();
140 
141  const bool status = auxlib::inv_sympd_rcond(out, is_sympd_junk, rcond_calc, rcond_threshold);
142 
143  if(status && arma_isfinite(rcond_calc)) { return true; }
144 
145  arma_extra_debug_print("op_pinv: sympd optimisation failed");
146  // auxlib::inv_sympd_rcond() will fail if A isn't really positive definite or its rcond is below rcond_threshold
147  }
148 
149  if(do_sym)
150  {
151  arma_extra_debug_print("op_pinv: symmetric/hermitian optimisation");
152 
153  return op_pinv::apply_sym(out, A, tol, method_id);
154  }
155 
156  return op_pinv::apply_gen(out, A, tol, method_id);
157  }

161 template<typename eT>
162 inline
163 bool
164 op_pinv::apply_diag(Mat<eT>& out, const Mat<eT>& A, typename get_pod_type<eT>::result tol)
165  {
166  arma_extra_debug_sigprint();
167 
168  typedef typename get_pod_type<eT>::result T;
169 
170  out.zeros(A.n_cols, A.n_rows);
171 
172  const uword N = (std::min)(A.n_rows, A.n_cols);
173 
174  podarray<T> diag_abs_vals(N);
175 
176  T max_abs_Aii = T(0);
177 
178  for(uword i=0; i<N; ++i)
179  {
180  const eT Aii = A.at(i,i);
181  const T abs_Aii = std::abs(Aii);
182 
183  if(arma_isnan(Aii)) { return false; }
184 
185  diag_abs_vals[i] = abs_Aii;
186 
187  max_abs_Aii = (abs_Aii > max_abs_Aii) ? abs_Aii : max_abs_Aii;
188  }
189 
190  if(tol == T(0)) { tol = (std::max)(A.n_rows, A.n_cols) * max_abs_Aii * std::numeric_limits<T>::epsilon(); }
191 
192  for(uword i=0; i<N; ++i)
193  {
194  if(diag_abs_vals[i] >= tol)
195  {
196  const eT Aii = A.at(i,i);
197 
198  if(Aii != eT(0)) { out.at(i,i) = eT(eT(1) / Aii); }
199  }
200  }
201 
202  return true;
203  }

208 inline
209 bool
210 op_pinv::apply_sym(Mat<eT>& out, const Mat<eT>& A, typename get_pod_type<eT>::result tol, const uword method_id)
211  {
212  arma_extra_debug_sigprint();
213 
214  typedef typename get_pod_type<eT>::result T;
215 
216  Col< T> eigval;
217  Mat<eT> eigvec;
218 
219  const bool status = ((method_id == uword(0)) || (method_id == uword(2))) ? auxlib::eig_sym_dc(eigval, eigvec, A) : auxlib::eig_sym(eigval, eigvec, A);
220 
221  if(status == false) { return false; }
222 
223  if(eigval.n_elem == 0) { out.zeros(A.n_cols, A.n_rows); return true; }
224 
225  Col<T> abs_eigval = arma::abs(eigval);
226 
227  const uvec indices = sort_index(abs_eigval, "descend");
228 
229  abs_eigval = abs_eigval.elem(indices);
230  eigval = eigval.elem(indices);
231  eigvec = eigvec.cols(indices);
232 
233  // set tolerance to default if it hasn't been specified
234  if(tol == T(0)) { tol = (std::max)(A.n_rows, A.n_cols) * abs_eigval[0] * std::numeric_limits<T>::epsilon(); }
235 
236  uword count = 0;
237 
238  for(uword i=0; i < abs_eigval.n_elem; ++i) { count += (abs_eigval[i] >= tol) ? uword(1) : uword(0); }
239 
240  if(count == 0) { out.zeros(A.n_cols, A.n_rows); return true; }
241 
242  Col<T> eigval2(count, arma_nozeros_indicator());
243 
244  uword count2 = 0;
245 
246  for(uword i=0; i < eigval.n_elem; ++i)
247  {
248  const T abs_val = abs_eigval[i];
249  const T val = eigval[i];
250 
251  if(abs_val >= tol) { eigval2[count2] = (val != T(0)) ? T(T(1) / val) : T(0); ++count2; }
252  }
253 
254  const Mat<eT> eigvec_use(eigvec.memptr(), eigvec.n_rows, count, false);
255 
256  out = (eigvec_use * diagmat(eigval2)).eval() * eigvec_use.t();
257 
258  return true;
259  }

264 template<typename eT>
265 inline
266 bool
267 op_pinv::apply_gen(Mat<eT>& out, Mat<eT>& A, typename get_pod_type<eT>::result tol, const uword method_id)
268  {
269  arma_extra_debug_sigprint();
270 
271  typedef typename get_pod_type<eT>::result T;
272 
273  const uword n_rows = A.n_rows;
274  const uword n_cols = A.n_cols;
275 
276  // economical SVD decomposition
277  Mat<eT> U;
278  Col< T> s;
279  Mat<eT> V;
280 
281  if(n_cols > n_rows) { A = trans(A); }
282 
283  const bool status = ((method_id == uword(0)) || (method_id == uword(2))) ? auxlib::svd_dc_econ(U, s, V, A) : auxlib::svd_econ(U, s, V, A, 'b');
284 
285  if(status == false) { return false; }
286 
287  // set tolerance to default if it hasn't been specified
288  if( (tol == T(0)) && (s.n_elem > 0) ) { tol = (std::max)(n_rows, n_cols) * s[0] * std::numeric_limits<T>::epsilon(); }
289 
290  uword count = 0;
291 
292  for(uword i=0; i < s.n_elem; ++i) { count += (s[i] >= tol) ? uword(1) : uword(0); }
293 
294  if(count == 0) { out.zeros(n_cols, n_rows); return true; }
295 
296  Col<T> s2(count, arma_nozeros_indicator());
297 
298  uword count2 = 0;
299 
300  for(uword i=0; i < s.n_elem; ++i)
301  {
302  const T val = s[i];
303 
304  if(val >= tol) { s2[count2] = (val > T(0)) ? T(T(1) / val) : T(0); ++count2; }
305  }
306 
307  const Mat<eT> U_use(U.memptr(), U.n_rows, count, false);
308  const Mat<eT> V_use(V.memptr(), V.n_rows, count, false);
309 
310  Mat<eT> tmp;
311 
312  if(n_rows >= n_cols)
313  {
314  // out = ( (V.n_cols > count) ? V.cols(0,count-1) : V ) * diagmat(s2) * trans( (U.n_cols > count) ? U.cols(0,count-1) : U );
315 
316  tmp = V_use * diagmat(s2);
317 
318  out = tmp * trans(U_use);
319  }
320  else
321  {
322  // out = ( (U.n_cols > count) ? U.cols(0,count-1) : U ) * diagmat(s2) * trans( (V.n_cols > count) ? V.cols(0,count-1) : V );
323 
324  tmp = U_use * diagmat(s2);
325 
326  out = tmp * trans(V_use);
327  }
328 
329  return true;
330  }
