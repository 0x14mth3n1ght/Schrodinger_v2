265 eT
266 op_median::direct_median(std::vector<eT>& X)
267  {
268  arma_extra_debug_sigprint();
269 
270  const uword n_elem = uword(X.size());
271  const uword half = n_elem/2;
272 
273  typename std::vector<eT>::iterator first = X.begin();
274  typename std::vector<eT>::iterator nth = first + half;
275  typename std::vector<eT>::iterator pastlast = X.end();
276 
277  std::nth_element(first, nth, pastlast);
278 
279  if((n_elem % 2) == 0) // even number of elements
280  {
281  typename std::vector<eT>::iterator start = X.begin();
282  typename std::vector<eT>::iterator pastend = start + half;
283 
284  const eT val1 = (*nth);
285  const eT val2 = (*(std::max_element(start, pastend)));
286 
287  return op_mean::robust_mean(val1, val2);
288  }
289  else // odd number of elements
290  {
291  return (*nth);
292  }
