26 inline
27 void
28 op_hist::apply_noalias(Mat<uword>& out, const Mat<eT>& A, const uword n_bins, const uword dim)
29  {
30  arma_extra_debug_sigprint();
31 
32  arma_debug_check( ((A.is_vec() == false) && (A.is_empty() == false)), "hist(): only vectors are supported when automatically determining bin centers" );
33 
34  if(n_bins == 0) { out.reset(); return; }
35 
36  uword A_n_elem = A.n_elem;
37  const eT* A_mem = A.memptr();
38 
39  eT min_val = priv::most_pos<eT>();
40  eT max_val = priv::most_neg<eT>();
41 
42  uword i,j;
43  for(i=0, j=1; j < A_n_elem; i+=2, j+=2)
44  {
45  const eT val_i = A_mem[i];
46  const eT val_j = A_mem[j];
47 
48  if(min_val > val_i) { min_val = val_i; }
49  if(min_val > val_j) { min_val = val_j; }
50 
51  if(max_val < val_i) { max_val = val_i; }
52  if(max_val < val_j) { max_val = val_j; }
53  }
54 
55  if(i < A_n_elem)
56  {
57  const eT val_i = A_mem[i];
58 
59  if(min_val > val_i) { min_val = val_i; }
60  if(max_val < val_i) { max_val = val_i; }
61  }
62 
63  if(min_val == max_val)
64  {
65  min_val -= (n_bins/2);
66  max_val += (n_bins/2);
67  }
68 
69  if(arma_isfinite(min_val) == false) { min_val = priv::most_neg<eT>(); }
70  if(arma_isfinite(max_val) == false) { max_val = priv::most_pos<eT>(); }
71 
72  Col<eT> c(n_bins, arma_nozeros_indicator());
73  eT* c_mem = c.memptr();
74 
75  for(uword ii=0; ii < n_bins; ++ii)
76  {
77  c_mem[ii] = (0.5 + ii) / double(n_bins);
78  }
79 
80  c = ((max_val - min_val) * c) + min_val;
81 
82  glue_hist::apply_noalias(out, A, c, dim);
83  }
